import pandas as pd

def calculate_roi(monthly_growth_rate, initial_revenue, months):
    final_revenue = initial_revenue * ((1 + monthly_growth_rate) ** months)
    absolute_increase = final_revenue - initial_revenue
    roi_percent = (absolute_increase / initial_revenue) * 100
    return absolute_increase, roi_percent

# 가정: 초기 월 매출 1 천만 원, 3 개월 성장률 5%, 최종 수익 증가율 계산

monthly_growth_rate = 0.05  # 5%
months = 3
abs_increase, roi_percent = calculate_roi(monthly_growth_rate, initial_revenue, months)

print(f"초기 매출: {initial_revenue:,} 원")
print(f"3 개월 후 매출: {(initial_revenue * ((1 + monthly_growth_rate) ** months):,} 원")
print(f"증가액: {abs_increase:,} 원")
print(f"ROI (증가율): {roi_percent:.2f}%")
