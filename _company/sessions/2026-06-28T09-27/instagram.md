## 📄 React 컴포넌트 코드 (간단한 예시)
```jsx
// TrustWidget.jsx
import { useState } from 'react';

function TrustWidget({ trustLevel }) {
  const colors = {
    green: '#2ECC71',
    yellow: '#FFC800',
    red: '#FF4B4B'
  };

  return (
    <div style={{ 
      color: colors[trustLevel], 
      padding: '1rem', 
      borderRadius: '8px' 
    }}>
      신뢰도: {trustLevel} ({colors[trustLevel]})
    </div>
  );
}
```
