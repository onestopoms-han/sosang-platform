        # 예시: API 응답 2 회 검증 로직
        def validate_paypal_response(response):
            if response.status_code != 200:
                return False
            # 1 번 검증
            data = json.loads(response.text)
            if 'error' in data:
                raise Exception("PayPal API Error")
            # 2 번 검증 (중복 데이터 필터링 등)
            return True
        ```
